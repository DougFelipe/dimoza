// Função principal do programa
Unit main() {
    // 1. Declara e inicializa as variáveis com os valores dados.
    //    Usamos o tipo Float, que é nativo da sua linguagem.
    Float x <- 3.0;
    Float y <- 2.5;
    Int c <- 10;

    // 2. Avalia a expressão x*x - y + c
    //    A sua linguagem suporta operações aritméticas diretas.
    //    O compilador gerará código C que lida com a conversão de tipos
    //    (Int para Float) automaticamente na expressão.
    Float resultado <- x * x - y + c;

    // 3. Imprime o resultado final na tela.
    print resultado;
}
